Measurement of Prediction Quality

Holt- Winters

Functions

ER <- function(pred, real){
  
  return (sum(abs(pred-real))/abs(sum(real)))
  
}

#mean square Error

ECM <- function(pred, real){
  
  N <- length(real)
  ss <- sum((real-pred)^2)
  return(ss/N)
  
}

PFA <- function(pred, real){
  total <- 0
  N <- length(pred)
  for (i in 1:N) {
    if(pred[i]>= real[i])
      total <- total +1
  }
  return(total/N)
}


PTFA <- function(pred, real){
  
  total <- 0
  sreal <- 0
  N <- length(pred)
  for (i in 1:N) {
    if(pred[i]>= real[i]){
      total <- total + (pred[i]-real[i])
      sreal <- sreal +abs(real[i])
      }
  }
  return(total/sreal)
}

Function to calibrate models

calibrate <- function(train, test){
  error.c <- Inf
  alpha.i <- 0.1
  while(alpha.i<=1){
    beta.i <- 0
    while(beta.i <=1){
    gamma.i <- 0
    while(gamma.i <=1){
      mod.i <- HoltWinters(train, alpha = alpha.i, beta = beta.i, gamma = gamma.i)
      pred.i <- predict(mod.i, n.ahead = length(test))
      error.i <- sqrt(ECM(pred.i, test))
      if(error.i < error.c){
        error.c <- error.i
        mod.c <- mod.i
      }
      gamma.i <- gamma.i + 0.1
      
      
    }
    beta.i <- beta.i+1
  }
    alpha.i <- alpha.i +1
  }
  
  return(mod.c)
  
}
